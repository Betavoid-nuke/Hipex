    <script>
        // --- GLOBAL STATE ---
        let scene, camera, renderer, controls, wallGroup;
        const shapes = []; // {id, points: [...], wallHeight, windows:[]}
        let selectedShape = null;
        const SNAP_POINT_RADIUS = 5;
        const SNAP_DISTANCE = 10;
        let scale = 1.0;
        const MIN_SCALE = 0.2;
        const MAX_SCALE = 5.0;

        // --- ACTION STATES ---
        let isDrawing = false;
        let isPannning = false;
        let isDraggingFromSnap = false;
        let dragStartState = {};
        let activeSnapTarget = null;
        let hoveredSnapPoint = null;
        
        // Canvas Panning
        let panStart = { x: 0, y: 0 };
        let canvasOffset = { x: 0, y: 0 };

        // --- DOM ELEMENTS ---
        const canvas = document.getElementById('drawing-canvas');
        const ctx = canvas.getContext('2d');
        const canvasContainer = document.getElementById('canvas-container');
        const wallEditor = document.getElementById('wall-editor');
        const editorCanvas = document.getElementById('editor-canvas');
        const editorCtx = editorCanvas.getContext('2d');
        const wallHeightInput = document.getElementById('wall-height-input');


        // --- 3D SCENE INITIALIZATION ---
        function init3D() {
            const container = document.getElementById('renderer-container');
            scene = new THREE.Scene();
            scene.background = new THREE.Color(0x000000); // Black background
            camera = new THREE.PerspectiveCamera(60, container.clientWidth / container.clientHeight, 1, 5000);
            camera.position.set(200, 250, 350);
            camera.lookAt(0, 0, 0);
            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(container.clientWidth, container.clientHeight);
            renderer.setPixelRatio(window.devicePixelRatio);
            renderer.shadowMap.enabled = true;
            container.appendChild(renderer.domElement);
            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.screenSpacePanning = true;

            const hemiLight = new THREE.HemisphereLight( 0xffffff, 0x444444, 1.2 );
            hemiLight.position.set( 0, 300, 0 );
            scene.add( hemiLight );

            const dirLight = new THREE.DirectionalLight(0xffffff, 1.0);
            dirLight.position.set(100, 200, 150);
            dirLight.castShadow = true;
            dirLight.shadow.camera.top = 200;
            dirLight.shadow.camera.bottom = -200;
            dirLight.shadow.camera.left = -200;
            dirLight.shadow.camera.right = 200;
            scene.add(dirLight);

            const groundGeo = new THREE.PlaneGeometry(2000, 2000);
            const groundMat = new THREE.MeshStandardMaterial({ color: 0x111827, roughness: 0.9 });
            const ground = new THREE.Mesh(groundGeo, groundMat);
            ground.rotation.x = -Math.PI / 2;
            ground.receiveShadow = true;
            scene.add(ground);
            
            const gridHelper = new THREE.GridHelper(2000, 100, 0x6b7280, 0x4b5563);
            scene.add(gridHelper);

            wallGroup = new THREE.Group();
            scene.add(wallGroup);
            animate();
        }

        function animate() {
            requestAnimationFrame(animate);
            controls.update();
            renderer.render(scene, camera);
        }

        // --- 2D CANVAS INITIALIZATION & DRAWING ---
        function init2D() {
            resizeCanvas();
            canvas.addEventListener('mousedown', onMouseDown);
            canvas.addEventListener('mousemove', onMouseMove);
            canvas.addEventListener('mouseup', onMouseUp);
            canvas.addEventListener('mouseleave', onMouseLeave);
            canvas.addEventListener('contextmenu', e => e.preventDefault());
            canvas.addEventListener('wheel', onCanvasWheel);
            canvas.addEventListener('dblclick', onDoubleClick);
        }

        function resizeCanvas() {
            const rect = canvasContainer.getBoundingClientRect();
            canvas.width = rect.width;
            canvas.height = rect.height;
            drawAllShapes();
        }
        
        function getSnapPoints(shape) {
            const points = [];
            for (let i = 0; i < shape.points.length; i++) {
                const p1 = shape.points[i];
                const p2 = shape.points[(i + 1) % shape.points.length];
                points.push({
                    x: (p1.x + p2.x) / 2,
                    y: (p1.y + p2.y) / 2,
                    parentId: shape.id
                });
            }
            return points;
        }

        function drawAllShapes() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.save();
            ctx.translate(canvasOffset.x, canvasOffset.y);
            ctx.scale(scale, scale);

            // Draw all walls (polygons)
            shapes.forEach(shape => {
                const isSelected = selectedShape && selectedShape.id === shape.id;
                ctx.strokeStyle = isSelected ? '#6366f1' : '#9ca3af';
                ctx.fillStyle = isSelected ? 'rgba(99, 102, 241, 0.3)' : 'rgba(75, 85, 99, 0.5)';
                ctx.lineWidth = isSelected ? 2 / scale : 1 / scale;
                
                ctx.beginPath();
                ctx.moveTo(shape.points[0].x, shape.points[0].y);
                for (let i = 1; i < shape.points.length; i++) {
                    ctx.lineTo(shape.points[i].x, shape.points[i].y);
                }
                ctx.closePath();
                ctx.fill();
                ctx.stroke();
            });

            // Draw snap points
            ctx.lineWidth = 2 / scale;
            shapes.forEach(shape => {
                const snapPoints = getSnapPoints(shape);
                snapPoints.forEach(p => {
                    const isSnapTarget = activeSnapTarget && activeSnapTarget.x === p.x && activeSnapTarget.y === p.y;
                    const isHovered = hoveredSnapPoint && hoveredSnapPoint.x === p.x && hoveredSnapPoint.y === p.y;
                    
                    if (isSnapTarget) ctx.fillStyle = '#22c55e';
                    else if (isHovered) ctx.fillStyle = '#6366f1';
                    else ctx.fillStyle = 'rgba(107, 114, 128, 0.5)';
                    
                    ctx.strokeStyle = '#1f2937';
                    ctx.beginPath();
                    ctx.arc(p.x, p.y, SNAP_POINT_RADIUS / scale, 0, 2 * Math.PI);
                    ctx.fill();
                    ctx.stroke();
                });
            });
            ctx.restore();
        }

        // --- MOUSE/TOOL LOGIC ---
        let startPoint = { x: 0, y: 0 };

        function getMousePos(evt, useScale = true) {
            const rect = canvas.getBoundingClientRect();
            const currentScale = useScale ? scale : 1;
            return {
                x: (evt.clientX - rect.left - canvasOffset.x) / currentScale,
                y: (evt.clientY - rect.top - canvasOffset.y) / currentScale
            };
        }

        function onMouseDown(e) {
            if (e.button === 2) {
                isPannning = true;
                panStart.x = e.clientX;
                panStart.y = e.clientY;
                canvas.style.cursor = 'grabbing';
                return;
            }
            if (e.button !== 0) return;
            
            if (hoveredSnapPoint) {
                isDraggingFromSnap = true;
                dragStartState = {
                    startPos: { ...hoveredSnapPoint },
                    parentShape: shapes.find(s => s.id === hoveredSnapPoint.parentId)
                };
                hoveredSnapPoint = null;
                return;
            }

            isDrawing = true;
            startPoint = getMousePos(e);
        }
        
        function onDoubleClick(e) {
            const mousePos = getMousePos(e);
            // Point-in-polygon test
            const clickedShape = shapes.find(shape => {
                let inside = false;
                for (let i = 0, j = shape.points.length - 1; i < shape.points.length; j = i++) {
                    const xi = shape.points[i].x, yi = shape.points[i].y;
                    const xj = shape.points[j].x, yj = shape.points[j].y;
                    const intersect = ((yi > mousePos.y) !== (yj > mousePos.y))
                        && (mousePos.x < (xj - xi) * (mousePos.y - yi) / (yj - yi) + xi);
                    if (intersect) inside = !inside;
                }
                return inside;
            });
            
            if (clickedShape) {
                selectedShape = clickedShape;
                openWallEditor(clickedShape);
            } else {
                selectedShape = null;
                closeWallEditor();
            }
            drawAllShapes();
        }

        function onCanvasWheel(e) {
            e.preventDefault();
            const zoomIntensity = 0.1;
            const wheel = e.deltaY < 0 ? 1 : -1;
            const zoom = Math.exp(wheel * zoomIntensity);
            
            const mousePos = getMousePos(e, false);
            const newScale = Math.max(MIN_SCALE, Math.min(MAX_SCALE, scale * zoom));
            canvasOffset.x = mousePos.x - (mousePos.x - canvasOffset.x) * (newScale / scale);
            canvasOffset.y = mousePos.y - (mousePos.y - canvasOffset.y) * (newScale / scale);
            
            scale = newScale;
            drawAllShapes();
        }

        function onMouseMove(e) {
            if (isPannning) {
                const dx = e.clientX - panStart.x;
                const dy = e.clientY - panStart.y;
                canvasOffset.x += dx;
                canvasOffset.y += dy;
                panStart.x = e.clientX;
                panStart.y = e.clientY;
                drawAllShapes();
                return;
            }

            let currentPos = getMousePos(e);
            activeSnapTarget = null;
            hoveredSnapPoint = null;

            if (!isDraggingFromSnap) {
                let foundHover = null;
                 for (const shape of shapes) {
                    const snapPoints = getSnapPoints(shape);
                    for (const point of snapPoints) {
                        const dx = currentPos.x - point.x;
                        const dy = currentPos.y - point.y;
                        if (dx * dx + dy * dy < (SNAP_POINT_RADIUS / scale)**2) {
                            foundHover = point;
                            break;
                        }
                    }
                    if(foundHover) break;
                }
                if (hoveredSnapPoint !== foundHover) {
                    hoveredSnapPoint = foundHover;
                    drawAllShapes();
                }
            }
            
            if (isDraggingFromSnap) {
                let closestSnap = null;
                let minSnapDist = (SNAP_DISTANCE / scale)**2;
                shapes.forEach(shape => {
                    if (shape.id === dragStartState.parentShape.id) return;
                    getSnapPoints(shape).forEach(p => {
                        const distSq = (p.x - currentPos.x)**2 + (p.y - currentPos.y)**2;
                        if (distSq < minSnapDist) {
                            minSnapDist = distSq;
                            closestSnap = p;
                        }
                    });
                });
                if (closestSnap) {
                    currentPos.x = closestSnap.x;
                    currentPos.y = closestSnap.y;
                    activeSnapTarget = closestSnap;
                }
            }
            
            drawAllShapes();
            ctx.save();
            ctx.translate(canvasOffset.x, canvasOffset.y);
            ctx.scale(scale, scale);
            ctx.fillStyle = 'rgba(99, 102, 241, 0.3)';
            ctx.strokeStyle = 'rgba(139, 142, 255, 0.8)';
            ctx.lineWidth = 1 / scale;

            if (isDrawing) {
                ctx.beginPath();
                ctx.rect(startPoint.x, startPoint.y, currentPos.x - startPoint.x, currentPos.y - startPoint.y);
                ctx.fill();
                ctx.stroke();
            } else if (isDraggingFromSnap) {
                const { startPos, parentShape } = dragStartState;
                const thickness = Math.min(parentShape.width, parentShape.height); // Simplified thickness
                
                const vec = { x: currentPos.x - startPos.x, y: currentPos.y - startPos.y };
                const len = Math.sqrt(vec.x * vec.x + vec.y * vec.y);
                if (len === 0) return;
                const perpVec = { x: -vec.y / len, y: vec.x / len };

                const p1 = { x: startPos.x - perpVec.x * thickness / 2, y: startPos.y - perpVec.y * thickness / 2 };
                const p2 = { x: startPos.x + perpVec.x * thickness / 2, y: startPos.y + perpVec.y * thickness / 2 };
                const p3 = { x: currentPos.x + perpVec.x * thickness / 2, y: currentPos.y + perpVec.y * thickness / 2 };
                const p4 = { x: currentPos.x - perpVec.x * thickness / 2, y: currentPos.y - perpVec.y * thickness / 2 };

                ctx.beginPath();
                ctx.moveTo(p1.x, p1.y);
                ctx.lineTo(p2.x, p2.y);
                ctx.lineTo(p3.x, p3.y);
                ctx.lineTo(p4.x, p4.y);
                ctx.closePath();
                ctx.fill();
                ctx.stroke();
            }
            ctx.restore();
        }

        function onMouseUp(e) {
            if (isPannning) {
                isPannning = false;
                return;
            }

            let endPoint = getMousePos(e);
            
            if (isDraggingFromSnap && activeSnapTarget) {
                endPoint.x = activeSnapTarget.x;
                endPoint.y = activeSnapTarget.y;
            }

            if (isDrawing) {
                const rect = {
                    x: Math.min(startPoint.x, endPoint.x),
                    y: Math.min(startPoint.y, endPoint.y),
                    w: Math.abs(startPoint.x - endPoint.x),
                    h: Math.abs(startPoint.y - endPoint.y)
                };
                if (rect.w > 2 && rect.h > 2) {
                    shapes.push({
                        id: Date.now(),
                        points: [
                            { x: rect.x, y: rect.y },
                            { x: rect.x + rect.w, y: rect.y },
                            { x: rect.x + rect.w, y: rect.y + rect.h },
                            { x: rect.x, y: rect.y + rect.h }
                        ],
                        wallHeight: 50,
                        windows: [],
                        width: rect.w, // Store for thickness calculation
                        height: rect.h
                    });
                }
            } else if (isDraggingFromSnap) {
                const { startPos, parentShape } = dragStartState;
                const thickness = Math.min(parentShape.width, parentShape.height);
                const vec = { x: endPoint.x - startPos.x, y: endPoint.y - startPos.y };
                const len = Math.sqrt(vec.x * vec.x + vec.y * vec.y);
                
                if (len > 2) {
                    const perpVec = { x: -vec.y / len, y: vec.x / len };
                    const p1 = { x: startPos.x - perpVec.x * thickness / 2, y: startPos.y - perpVec.y * thickness / 2 };
                    const p2 = { x: startPos.x + perpVec.x * thickness / 2, y: startPos.y + perpVec.y * thickness / 2 };
                    const p3 = { x: endPoint.x + perpVec.x * thickness / 2, y: endPoint.y + perpVec.y * thickness / 2 };
                    const p4 = { x: endPoint.x - perpVec.x * thickness / 2, y: endPoint.y - perpVec.y * thickness / 2 };
                    
                    const newWall = {
                        id: Date.now(),
                        points: [p1, p2, p3, p4],
                        wallHeight: parentShape.wallHeight,
                        windows: [],
                        width: len, // Store for thickness calculation
                        height: thickness
                    };
                    shapes.push(newWall);
                }
            }

            isDrawing = false;
            isDraggingFromSnap = false;
            activeSnapTarget = null;
            hoveredSnapPoint = null;
            drawAllShapes();
            update3DScene();
        }

        function onMouseLeave() {
            isDrawing = false;
            isPannning = false;
            isDraggingFromSnap = false;
            activeSnapTarget = null;
            hoveredSnapPoint = null;
            drawAllShapes();
        }
        
        // --- WALL EDITOR --- (Simplified, no canvas for now)
        function openWallEditor(shape) {
            wallEditor.style.display = 'flex';
            wallHeightInput.value = shape.wallHeight;
            // Disabling editor canvas for angled walls for now
            document.getElementById('editor-canvas').style.display = 'none';
            document.getElementById('editor-tool-window').style.display = 'none';
        }

        function closeWallEditor() {
            wallEditor.style.display = 'none';
            selectedShape = null;
        }

        // --- 3D MODELING ---
        function update3DScene() {
            while (wallGroup.children.length > 0) {
                wallGroup.remove(wallGroup.children[0]);
            }

            const wallMaterial = new THREE.MeshStandardMaterial({ color: 0xe5e7eb, roughness: 0.6, metalness: 0.1 });
            const offsetX = -canvas.width / 2;
            const offsetZ = -canvas.height / 2;

            shapes.forEach(shape => {
                const wallShape = new THREE.Shape();
                wallShape.moveTo(shape.points[0].x + offsetX, shape.points[0].y + offsetZ);
                for (let i = 1; i < shape.points.length; i++) {
                    wallShape.lineTo(shape.points[i].x + offsetX, shape.points[i].y + offsetZ);
                }
                
                const extrudeSettings = {
                    steps: 1,
                    depth: shape.wallHeight,
                    bevelEnabled: false
                };
                const geometry = new THREE.ExtrudeGeometry(wallShape, extrudeSettings);
                const wallMesh = new THREE.Mesh(geometry, wallMaterial.clone());
                
                wallMesh.rotation.x = -Math.PI / 2;
                wallMesh.position.y = shape.wallHeight;

                wallMesh.castShadow = true;
                wallMesh.receiveShadow = true;
                wallGroup.add(wallMesh);
            });
        }

        // --- UI AND EVENT LISTENERS ---
        window.addEventListener('resize', () => {
            resizeCanvas();
            const container = document.getElementById('renderer-container');
            camera.aspect = container.clientWidth / container.clientHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(container.clientWidth, container.clientHeight);
            update3DScene();
        });

        document.getElementById('tool-clear').addEventListener('click', () => {
            shapes.length = 0;
            selectedShape = null;
            drawAllShapes();
            update3DScene();
            closeWallEditor();
        });

        document.getElementById('close-editor-btn').addEventListener('click', () => {
            closeWallEditor();
            drawAllShapes();
        });
        document.getElementById('editor-tool-delete').addEventListener('click', () => {
            if (!selectedShape) return;
            const index = shapes.findIndex(s => s.id === selectedShape.id);
            if (index > -1) {
                shapes.splice(index, 1);
            }
            closeWallEditor();
            drawAllShapes();
            update3DScene();
        });
        wallHeightInput.addEventListener('change', () => {
            if (selectedShape) {
                selectedShape.wallHeight = parseFloat(wallHeightInput.value) || 50;
                update3DScene();
            }
        });

        // --- MAIN EXECUTION ---
        init2D();
        init3D();
        update3DScene();
    </script>
